--[[
  Shahd Queen - Radio Controller (Compact Version)
  Features: Minimize, Close, Auto-Hide, Spam System
  Organization: Dogs of Captivity (كلاب الأسر) 🐾
]]

local Players = game:GetService("Players")
local localPlayer = Players.LocalPlayer
local spamming = false

-- وظيفة البحث عن الريموت
local function getRemote()
    for _, v in pairs(localPlayer.PlayerGui:GetDescendants()) do
        if v.Name == "Remote" and v:IsA("RemoteEvent") and v.Parent.Name == "MountedGui" then
            return v
        end
    end
    return nil
end

local remote = getRemote()

if remote then
    local sg = Instance.new("ScreenGui", game:GetService("CoreGui"))
    sg.Name = "ShahdCompactControl"

    -- اللوحة الرئيسية
    local mainFrame = Instance.new("Frame", sg)
    mainFrame.Size = UDim2.new(0, 450, 0, 40) -- تبدأ صغيرة (الهيدر فقط)
    mainFrame.Position = UDim2.new(0.5, -225, 0.1, 0)
    mainFrame.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
    mainFrame.Active = true
    mainFrame.Draggable = true
    mainFrame.ClipsDescendants = true -- لإخفاء المحتوى عند التصغير
    Instance.new("UICorner", mainFrame)

    -- الهيدر
    local header = Instance.new("Frame", mainFrame)
    header.Size = UDim2.new(1, 0, 0, 40)
    header.BackgroundColor3 = Color3.fromRGB(255, 105, 180)
    Instance.new("UICorner", header)

    local title = Instance.new("TextLabel", header)
    title.Size = UDim2.new(1, -80, 1, 0)
    title.Position = UDim2.new(0, 10, 0, 0)
    title.Text = "shhode320 Control 👑"
    title.TextColor3 = Color3.new(1, 1, 1)
    title.BackgroundTransparency = 1
    title.Font = Enum.Font.FredokaOne; title.TextSize = 18; title.TextXAlignment = Enum.TextXAlignment.Left

    -- زر الإغلاق (X)
    local closeBtn = Instance.new("TextButton", header)
    closeBtn.Size = UDim2.new(0, 30, 0, 30); closeBtn.Position = UDim2.new(1, -35, 0, 5)
    closeBtn.Text = "X"; closeBtn.BackgroundColor3 = Color3.fromRGB(200, 0, 0); closeBtn.TextColor3 = Color3.new(1, 1, 1)
    Instance.new("UICorner", closeBtn)
    closeBtn.MouseButton1Click:Connect(function() sg:Destroy() end)

    -- زر التصغير/التكبير (-)
    local minBtn = Instance.new("TextButton", header)
    minBtn.Size = UDim2.new(0, 30, 0, 30); minBtn.Position = UDim2.new(1, -70, 0, 5)
    minBtn.Text = "+"; minBtn.BackgroundColor3 = Color3.fromRGB(40, 40, 40); minBtn.TextColor3 = Color3.new(1, 1, 1)
    Instance.new("UICorner", minBtn)

    local isMinimized = true
    minBtn.MouseButton1Click:Connect(function()
        isMinimized = not isMinimized
        mainFrame:TweenSize(isMinimized and UDim2.new(0, 450, 0, 40) or UDim2.new(0, 450, 0, 400), "Out", "Quart", 0.3, true)
        minBtn.Text = isMinimized and "+" or "-"
    end)

    -- محتوى اللوحة (القائمة والأزرار)
    local contentFrame = Instance.new("Frame", mainFrame)
    contentFrame.Size = UDim2.new(1, 0, 0, 360); contentFrame.Position = UDim2.new(0, 0, 0, 40); contentFrame.BackgroundTransparency = 1

    local playerList = Instance.new("ScrollingFrame", contentFrame)
    playerList.Size = UDim2.new(0.4, 0, 0.9, 0); playerList.Position = UDim2.new(0.05, 0, 0.05, 0)
    playerList.BackgroundColor3 = Color3.fromRGB(25, 25, 25); playerList.CanvasSize = UDim2.new(0, 0, 10, 0)
    Instance.new("UIListLayout", playerList).Padding = UDim.new(0, 5); Instance.new("UICorner", playerList)

    local controlSide = Instance.new("Frame", contentFrame)
    controlSide.Size = UDim2.new(0.45, 0, 0.9, 0); controlSide.Position = UDim2.new(0.5, 0, 0.05, 0); controlSide.BackgroundTransparency = 1

    local targetLabel = Instance.new("TextLabel", controlSide)
    targetLabel.Size = UDim2.new(1, 0, 0, 30); targetLabel.Text = "المستهدف: الكل"; targetLabel.BackgroundColor3 = Color3.fromRGB(40, 40, 40); targetLabel.TextColor3 = Color3.new(1,1,1)

    local codeInput = Instance.new("TextBox", controlSide)
    codeInput.Size = UDim2.new(1, 0, 0, 50); codeInput.Position = UDim2.new(0, 0, 0.15, 0); codeInput.Text = "140217738705613"
    codeInput.BackgroundColor3 = Color3.fromRGB(30, 30, 30); codeInput.TextColor3 = Color3.new(1,1,1); codeInput.TextScaled = true

    local selectedTarget = "الكل"

    local function fireRemote(target)
        if target == "الكل" then
            for _, p in pairs(Players:GetPlayers()) do remote:FireServer(p, codeInput.Text) end
        else
            local p = Players:FindFirstChild(target)
            if p then remote:FireServer(p, codeInput.Text) end
        end
    end

    local playBtn = Instance.new("TextButton", controlSide)
    playBtn.Size = UDim2.new(1, 0, 0, 40); playBtn.Position = UDim2.new(0, 0, 0.35, 0); playBtn.Text = "تشغيل ▶️"; playBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0); playBtn.TextColor3 = Color3.new(1, 1, 1)

    local spamBtn = Instance.new("TextButton", controlSide)
    spamBtn.Size = UDim2.new(1, 0, 0, 40); spamBtn.Position = UDim2.new(0, 0, 0.5, 0); spamBtn.Text = "سبام 🚀"; spamBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50); spamBtn.TextColor3 = Color3.new(1, 1, 1)

    local function refreshList()
        for _, child in pairs(playerList:GetChildren()) do if child:IsA("TextButton") then child:Destroy() end end
        local allBtn = Instance.new("TextButton", playerList)
        allBtn.Size = UDim2.new(1, -10, 0, 35); allBtn.Text = "الكل (Global)"; allBtn.BackgroundColor3 = Color3.fromRGB(255, 105, 180)
        allBtn.MouseButton1Click:Connect(function() selectedTarget = "الكل"; targetLabel.Text = "المستهدف: الكل" end)
        for _, p in pairs(Players:GetPlayers()) do
            local pBtn = Instance.new("TextButton", playerList)
            pBtn.Size = UDim2.new(1, -10, 0, 35); pBtn.Text = p.Name; pBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45); pBtn.TextColor3 = Color3.new(1,1,1)
            pBtn.MouseButton1Click:Connect(function() selectedTarget = p.Name; targetLabel.Text = "المستهدف: " .. p.Name end)
        end
    end

    playBtn.MouseButton1Click:Connect(function() fireRemote(selectedTarget) end)
    spamBtn.MouseButton1Click:Connect(function()
        spamming = not spamming; spamBtn.Text = spamming and "STOP 🛑" or "سبام 🚀"
        spamBtn.BackgroundColor3 = spamming and Color3.fromRGB(150, 0, 0) or Color3.fromRGB(50, 50, 50)
        while spamming do fireRemote(selectedTarget); task.wait(0.5) end
    end)

    refreshList(); Players.PlayerAdded:Connect(refreshList); Players.PlayerRemoving:Connect(refreshList)
else
    print("❌ الريموت غير موجود. تأكدي من لبس الراديو!")
end
