-- المتغيرات البرمجية
local isRainbow = false
local currentRainbowThread = nil

-- وظيفة قوس قزح النيون
local function startRainbow()
    if currentRainbowThread then task.cancel(currentRainbowThread) end
    currentRainbowThread = task.spawn(function()
        while isRainbow do
            local hue = (tick() * 1.5) % 1 -- سرعة التدرج
            local neonColor = Color3.fromHSV(hue, 0.9, 1)
            pcall(function()
                applyTitle:FireServer(textInput.Text, neonColor)
            end)
            task.wait(0.1)
        end
    end)
end

-- برمجة الأزرار
rainbowBtn.MouseButton1Click:Connect(function()
    isRainbow = not isRainbow
    if isRainbow then
        rainbowBtn.Text = "إيقاف التدرج (الوضع النشط) ✅"
        rainbowBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
        startRainbow()
    else
        rainbowBtn.Text = "تفعيل تدرج النيون (Rainbow) 🌈"
        rainbowBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    end
end)

colorBtn.MouseButton1Click:Connect(function()
    isRainbow = false -- إيقاف الرينبو عند اختيار لون ثابت
    rainbowBtn.Text = "تفعيل تدرج النيون (Rainbow) 🌈"
    rainbowBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    
    -- تطبيق اللون الحالي (يمكنك تعديل RGB يدوياً هنا)
    local manualColor = Color3.fromRGB(255, 0, 127) -- وردي فاقع كمثال
    applyTitle:FireServer(textInput.Text, manualColor)
end)
